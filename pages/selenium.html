1. Introduction to Selenium

Selenium is the most popular open-source framework for automated web application testing. It provides tools and libraries for automating browser-based applications across different platforms and programming languages.

Selenium Components

  • Selenium WebDriver: Programmatic interface for controlling browsers
  • Selenium IDE: Browser extension for record and playback
  • Selenium Grid: Server for running tests in parallel

Supported Browsers

  • Chrome / Chromium
  • Firefox
  • Safari
  • Edge
  • Internet Explorer (legacy)

3. Basic WebDriver Operations

Launching and Closing Browser

from selenium import webdriver from selenium.webdriver.chrome.service import Service # Simple way driver = webdriver.Chrome() # With explicit path service = Service("/path/to/chromedriver") driver = webdriver.Chrome(service=service) # With options from selenium.webdriver.chrome.options import Options options = Options() options.add_argument("--headless") # Run in headless mode options.add_argument("--disable-gpu") driver = webdriver.Chrome(options=options) # Close specific tab driver.close() # Close entire browser driver.quit()

Navigation

from selenium import webdriver driver = webdriver.Chrome() # Navigate to URL driver.get("https://example.com") # Get current URL current_url = driver.current_url # Get page title title = driver.title # Navigate back driver.back() # Navigate forward driver.forward() # Refresh page driver.refresh()

Window Management

# Maximize window driver.maximize_window() # Set window size driver.set_window_size(1024, 768) # Get window size size = driver.get_window_size() # Get window position position = driver.get_window_position() # Switch between windows all_windows = driver.window_handles driver.switch_to.window(all_windows[1]) # Switch to alert alert = driver.switch_to.alert

4. Locating Elements

Locator Strategies

from selenium.webdriver.common.by import By # ID element = driver.find_element(By.ID, "element_id") # Class Name element = driver.find_element(By.CLASS_NAME, "class_name") # CSS Selector element = driver.find_element(By.CSS_SELECTOR, "div.class > span") # XPath element = driver.find_element(By.XPATH, "//button[@class='login']") # Name element = driver.find_element(By.NAME, "username") # Link Text element = driver.find_element(By.LINK_TEXT, "Click Here") # Partial Link Text element = driver.find_element(By.PARTIAL_LINK_TEXT, "Click") # Tag Name elements = driver.find_elements(By.TAG_NAME, "button")

Finding Multiple Elements

from selenium.webdriver.common.by import By # Find all elements all_buttons = driver.find_elements(By.TAG_NAME, "button") # Iterate through elements for button in all_buttons: print(button.text) # Find elements relative to parent parent = driver.find_element(By.CLASS_NAME, "container") children = parent.find_elements(By.CLASS_NAME, "item")

XPath Examples

# By text element = driver.find_element(By.XPATH, "//button[text()='Submit']") # By partial text element = driver.find_element(By.XPATH, "//button[contains(text(), 'Sub')]") # By attribute element = driver.find_element(By.XPATH, "//input[@type='email']") # By multiple attributes element = driver.find_element(By.XPATH, "//input[@type='text'][@name='username']") # By attribute value partial match element = driver.find_element(By.XPATH, "//div[contains(@class, 'error')]") # Parent element parent = driver.find_element(By.XPATH, "//input[@id='email']/parent::form") # Following sibling sibling = driver.find_element(By.XPATH, "//label[text()='Email']/following-sibling::input") # Index third_button = driver.find_element(By.XPATH, "//button[3]")

5. Waits and Explicit Waits

Implicit Wait

from selenium import webdriver driver = webdriver.Chrome() # Set implicit wait (applies to all find_element calls) driver.implicitly_wait(10) # 10 seconds # Now all element searches will wait up to 10 seconds element = driver.find_element(By.ID, "dynamic_element")

Explicit Wait (WebDriverWait)

from selenium.webdriver.support.ui import WebDriverWait from selenium.webdriver.support import expected_conditions as EC from selenium.webdriver.common.by import By # Create WebDriverWait instance wait = WebDriverWait(driver, 10) # Wait for element to be visible element = wait.until( EC.presence_of_element_located((By.ID, "element_id")) ) # Wait for element to be clickable button = wait.until( EC.element_to_be_clickable((By.CLASS_NAME, "submit-btn")) ) # Wait for text in element wait.until( EC.text_to_be_present_in_element( (By.ID, "message"), "Success" ) ) # Wait for element invisibility wait.until( EC.invisibility_of_element_located((By.CLASS_NAME, "loading")) )

Common Expected Conditions

from selenium.webdriver.support import expected_conditions as EC # Element is present in DOM EC.presence_of_element_located(locator) # Element is visible EC.visibility_of_element_located(locator) # Element is clickable EC.element_to_be_clickable(locator) # Element is selected EC.element_to_be_selected(locator) # Text present in element EC.text_to_be_present_in_element(locator, text) # Element has specific attribute value EC.element_has_css_class(element, class_name) # URL matches EC.url_contains("example.com") # All elements present EC.presence_of_all_elements_located(locator)

6. Advanced WebDriver Actions

Interactions with Elements

from selenium.webdriver.common.action_chains import ActionChains # Click element element = driver.find_element(By.ID, "button") element.click() # Double click action = ActionChains(driver) action.double_click(element).perform() # Right click action.context_click(element).perform() # Hover over element action.move_to_element(element).perform() # Drag and drop source = driver.find_element(By.ID, "draggable") target = driver.find_element(By.ID, "droppable") action.drag_and_drop(source, target).perform() # Enter text input_field = driver.find_element(By.ID, "email") input_field.send_keys("user@example.com") # Clear text input_field.clear() # Submit form form = driver.find_element(By.TAG_NAME, "form") form.submit()

Keyboard and Mouse Actions

from selenium.webdriver.common.keys import Keys from selenium.webdriver.common.action_chains import ActionChains # Type text with special keys input_field = driver.find_element(By.ID, "search") input_field.send_keys("selenium", Keys.ENTER) # Select all and delete input_field.send_keys(Keys.CONTROL + "a") input_field.send_keys(Keys.DELETE) # Tab to next field input_field.send_keys(Keys.TAB) # Copy, paste action = ActionChains(driver) action.key_down(Keys.CONTROL).send_keys("c").key_up(Keys.CONTROL).perform() action.key_down(Keys.CONTROL).send_keys("v").key_up(Keys.CONTROL).perform() # Release key action.key_up(Keys.CONTROL).perform()

Handling Alerts, Windows, and Frames

# Handle JavaScript Alert alert = driver.switch_to.alert alert.accept() # Click OK alert.dismiss() # Click Cancel alert.send_keys("text") # Enter text in prompt alert_text = alert.text # Get alert message # Switch to iframe iframe = driver.find_element(By.TAG_NAME, "iframe") driver.switch_to.frame(iframe) # Switch back to main content driver.switch_to.default_content() # Switch to window windows = driver.window_handles driver.switch_to.window(windows[1]) # Get/Switch to active element active = driver.switch_to.active_element active.send_keys("text")

JavaScript Execution

# Execute JavaScript result = driver.execute_script("return 1 + 1") print(result) # 2 # Get element property element = driver.find_element(By.ID, "element") value = driver.execute_script("return arguments[0].value", element) # Scroll to element driver.execute_script("arguments[0].scrollIntoView();", element) # Set element style driver.execute_script("arguments[0].style.display = 'block';", element) # Get all local storage local_storage = driver.execute_script("return window.localStorage")

7. Practical Examples

Login Test with Page Object Model

from selenium import webdriver from selenium.webdriver.common.by import By from selenium.webdriver.support.ui import WebDriverWait from selenium.webdriver.support import expected_conditions as EC class LoginPage: def __init__(self, driver): self.driver = driver self.wait = WebDriverWait(driver, 10) # Locators USERNAME_INPUT = (By.ID, "username") PASSWORD_INPUT = (By.ID, "password") LOGIN_BUTTON = (By.CLASS_NAME, "login-btn") ERROR_MESSAGE = (By.CLASS_NAME, "error-msg") # Methods def enter_username(self, username): element = self.wait.until( EC.presence_of_element_located(self.USERNAME_INPUT) ) element.clear() element.send_keys(username) def enter_password(self, password): element = self.wait.until( EC.presence_of_element_located(self.PASSWORD_INPUT) ) element.clear() element.send_keys(password) def click_login(self): button = self.wait.until( EC.element_to_be_clickable(self.LOGIN_BUTTON) ) button.click() def get_error_message(self): message = self.wait.until( EC.visibility_of_element_located(self.ERROR_MESSAGE) ) return message.text def login(self, username, password): self.enter_username(username) self.enter_password(password) self.click_login() # Test using Page Object import pytest class TestLogin: @pytest.fixture(autouse=True) def setup_teardown(self): self.driver = webdriver.Chrome() self.driver.get("https://example.com/login") yield self.driver.quit() def test_valid_login(self): login_page = LoginPage(self.driver) login_page.login("testuser@example.com", "Password123") assert "Dashboard" in self.driver.title def test_invalid_password(self): login_page = LoginPage(self.driver) login_page.login("testuser@example.com", "WrongPassword") error = login_page.get_error_message() assert "Invalid credentials" in error

E-Commerce Checkout Test

from selenium import webdriver from selenium.webdriver.common.by import By from selenium.webdriver.support.ui import WebDriverWait, Select from selenium.webdriver.support import expected_conditions as EC class ShoppingTest: def __init__(self): self.driver = webdriver.Chrome() self.wait = WebDriverWait(self.driver, 10) def test_complete_purchase(self): # Navigate to shop self.driver.get("https://shop.example.com") # Search for product search_box = self.driver.find_element(By.ID, "search") search_box.send_keys("laptop") search_box.send_keys(Keys.ENTER) # Select first product product = self.wait.until( EC.element_to_be_clickable( (By.XPATH, "//div[@class='product'][1]") ) ) product.click() # Add to cart add_btn = self.driver.find_element(By.CLASS_NAME, "add-to-cart") add_btn.click() # Go to cart cart_link = self.driver.find_element(By.LINK_TEXT, "Cart") cart_link.click() # Proceed to checkout checkout_btn = self.driver.find_element( By.CLASS_NAME, "checkout-btn" ) checkout_btn.click() # Fill shipping details first_name = self.driver.find_element(By.ID, "first_name") first_name.send_keys("John") last_name = self.driver.find_element(By.ID, "last_name") last_name.send_keys("Doe") address = self.driver.find_element(By.ID, "address") address.send_keys("123 Main Street") # Select country country_select = Select( self.driver.find_element(By.ID, "country") ) country_select.select_by_value("US") # Select shipping method shipping_options = self.driver.find_elements( By.NAME, "shipping_method" ) shipping_options[1].click() # Select second option # Fill payment details card_number = self.driver.find_element(By.ID, "card_number") card_number.send_keys("4532123456789010") # Submit order submit_btn = self.driver.find_element(By.ID, "submit_order") submit_btn.click() # Verify order confirmation self.wait.until( EC.visibility_of_element_located( (By.CLASS_NAME, "order-confirmation") ) ) assert "Order Confirmed" in self.driver.page_source self.driver.quit() # Run test test = ShoppingTest() test.test_complete_purchase()

8. Best Practices

Key Guidelines

  • Use Page Object Model: Encapsulate page elements and actions
  • Explicit Waits: Use WebDriverWait instead of sleep()
  • Unique Locators: Prefer ID and Name over CSS/XPath
  • Error Handling: Handle NoSuchElementException gracefully
  • Headless Mode: Use for CI/CD pipelines
  • Close Driver: Always quit driver in teardown

Avoiding Common Pitfalls

# ‚ùå Don't: Hard sleep import time time.sleep(5) # ‚úÖ Do: Use WebDriverWait from selenium.webdriver.support import expected_conditions as EC wait.until(EC.presence_of_element_located((By.ID, "element"))) # ‚ùå Don't: Fragile XPath driver.find_element(By.XPATH, "/html/body/div[1]/div[2]/form/input[3]") # ‚úÖ Do: More specific locators driver.find_element(By.ID, "submit_button") driver.find_element(By.CLASS_NAME, "submit-btn") # ‚ùå Don't: Catch all exceptions try: element = driver.find_element(By.ID, "element") except: pass # ‚úÖ Do: Catch specific exceptions from selenium.common.exceptions import NoSuchElementException try: element = driver.find_element(By.ID, "element") except NoSuchElementException: print("Element not found")
